# jQuery Plugin vzv_sub_menu
add sub menu from content
jQuery(document).ready(function($){
      $('body').vzv_sub_menu({
          'headers' : 'h2, h3',
          'after': true,
          'speed': 800,
          'title': 'Content NAV:',
      });
});

# See demo(https://vadya8z.github.io/vzv_sub_menu/)
